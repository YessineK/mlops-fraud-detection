pipeline {
    agent any

    environment {
        MLFLOW_TRACKING_USERNAME = 'karrayyessine1'
        MLFLOW_TRACKING_PASSWORD = credentials('dagshub-token')
        MLFLOW_TRACKING_URI = 'https://dagshub.com/karrayyessine1/mlops-fraud-detection.mlflow'
        DAGSHUB_TOKEN = credentials('dagshub-token')
        DAGSHUB_USER = 'karrayyessine1'
        DAGSHUB_REPO = 'mlops-fraud-detection'
    }

    stages {

        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Setup Environment') {
            steps {
                sh '''
                    echo "Creating Python virtual environment..."
                    python3 -m venv venv
                    source venv/bin/activate

                    pip install --upgrade pip
                    pip install -r requirements.txt
                    pip install dagshub
                '''
            }
        }

        stage('Continuous Training') {
            when { expression { false } }   // Désactivé volontairement
            steps {
                sh '''
                    echo "Starting Model Training..."
                    source venv/bin/activate
                    python3 Jenkins/train_model.py
                '''
            }
        }

        stage('Register Best Model') {
            when { expression { false } }
            steps {
                sh '''
                    echo "Registering Best Model..."
                    source venv/bin/activate
                    python3 Jenkins/register_best_model.py
                '''
            }
        }

        stage('Run Tests') {
            when { expression { false } }
            steps {
                sh '''
                    echo "Running Fraud Tests..."
                    source venv/bin/activate
                    python3 testing/test_fraud_scenario.py
                '''
            }
        }

        stage('Build and Deploy') {
            steps {
                sh '''
                    echo "Building & Deploying containers..."
                    docker compose down || true
                    docker compose up -d --build
                '''
            }
        }
    }

    post {
        always {
            cleanWs()
        }
        success {
            echo 'Pipeline completed successfully.'
        }
        failure {
            echo 'Pipeline failed.'
        }
    }
}
